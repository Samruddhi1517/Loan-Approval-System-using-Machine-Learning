import pandas as pd

# Load dataset
df = pd.read_csv("loanapproval.csv")

# Separate classes
approved = df[df['loan_approved'] == 1]
not_approved = df[df['loan_approved'] == 0]

# Randomly remove 450 records from approved class
approved_reduced = approved.sample(n=len(approved) - 450, random_state=42)

# Combine back with not approved
df_balanced = pd.concat([approved_reduced, not_approved])

# Shuffle dataset
df_balanced = df_balanced.sample(frac=1, random_state=42).reset_index(drop=True)

# Check new distribution
print(df_balanced['loan_approved'].value_counts())
